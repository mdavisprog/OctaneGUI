set(TARGET CodeEdit)

add_executable(
    ${TARGET}
    Main.cpp
)

target_include_directories(
    ${TARGET}
    PUBLIC ${OctaneGUI_INCLUDE}
    PUBLIC ${FRONTEND_INCLUDE}
)

target_link_libraries(
    ${TARGET}
    Frontend
    OctaneGUI
)

set_target_properties(
    ${TARGET}
    PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${BIN_DIR}
    RUNTIME_OUTPUT_DIRECTORY_DEBUG ${BIN_DIR}
    RUNTIME_OUTPUT_DIRECTORY_RELEASE ${BIN_DIR}
)

add_custom_command(
    OUTPUT ${BIN_DIR}/CodeEdit.json
    COMMAND ${CMAKE_COMMAND} -E copy ${CMAKE_CURRENT_SOURCE_DIR}/CodeEdit.json ${BIN_DIR}
    DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/CodeEdit.json
)

add_custom_target(
    ${TARGET}_Json ALL
    DEPENDS ${BIN_DIR}/CodeEdit.json
)

add_dependencies(
    ${TARGET}
    ${TARGET}_Json
)
